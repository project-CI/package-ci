 13 /**
 14  * Can we retrieve pets by their ID numbers?
 15  * @name obtainPetsByIds
 16  * @memberof EndToEndTests
 17  * @function
 18  * @param  null
 19  * @return {None}
 20  */
 21 module.exports = {
 22   'Step Through Pets': function obtainPetsByIds(client) {
 23     client
 24       .url('http://localhost:3000')
 25       .waitForElementVisible('body', 1000)
 26       .verify.elementPresent('button[id=nextPet]')
 27       .verify.elementPresent('p[id=petNote]')
 28       .assert.containsText('p[id=petNote]', '6133627026')
 29       .assert.containsText('p[id=petNote]', 'unknown')
 30       .click('button[id=nextPet]')
 31       .pause(3000)
 32       .assert.containsText('p[id=petNote]', '6133627027')
 33       .assert.containsText('p[id=petNote]', 'weasel')
 34 
 35       .end();
 36   },
 37 };
