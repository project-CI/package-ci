 10 Session.setDefault('idPet', 6133627026);
 11 Session.setDefault('pet', { name: 'unknown'});
 12 
 13 Template.petStoreTest.helpers({
 14 
 15   /**
 16    * petStoreTest template's helper for getting a pet ID
 17    * @return {Integer} A valid pet ID
 18    */
 19   idPet: function getCounter() {
 20     return Session.get('idPet');
 21   },
 22   /**
 23    * petStoreTest template's helper for getting a pet name as returned from the API
 24    * @return {String} The pet's name according to the remote API.
 25    */
 26   namePet: function getPetName() {
 27     return Session.get('pet').name;
 28   },
 29 
 30 });
 31 
 32 Template.petStoreTest.events({
 33   /**
 34    * Handle clicks of the button to step through a short list of pets
 35    * @return {null} None
 36    */
 37   'click button': function clickMe() {
 38     Session.set('idPet', Session.get('idPet') + 1);
 39 
 40     Meteor.call('package-ci_getPetById', Session.get('idPet'), function setPet(err, thePet) {
 41       if (err) {
 42         console.log('Server error :: ' + err);
 43         Session.set('pet', { name: 'error'});
 44       } else {
 45         Session.set('pet', thePet);
 46       }
 47     });
 48   },
 49 });
