 21 /**
 22  * @constant
 23  * @description  An instance of the NodeJS module 'swagger-client'
 24  * Used to instaniate proxies of remote REST APIs that
 25  * publish Swagger format specifications
 26  * @type swagger-client
 27  * @memberof PetStore
 28  * @inner
 29  */
 30 const Client = Npm.require('swagger-client');
 31 
 32 /**
 33  * @constant
 34  * @description  Text representation of the URL of the REST API this package will proxy.
 35  * @type String
 36  * @memberof PetStore
 37  * @inner
 38  */
 39 const swaggerSpecURL = 'http://petstore.swagger.io/v2/swagger.json';
 40 
 41 /**
 42  * Instantiates the PetStore namespace proxy of the Swagger projects demo API
 43  * @name getSwaggerProxy
 44  * @memberof PetStore
 45  * @function
 46  * @param  {string} swaggerSpecURL
 47  * @returns {PetStore}
 48  */
 49 const getSwaggerProxy = Meteor.wrapAsync(
 50   function wrpr(swaggerSpec, callback) {
 51     var prxySwagger = new Client({
 52       url: swaggerSpec,
 53       success: function suxs() { callback(null, prxySwagger); },
 54       error: function errs() { callback(null, prxySwagger); },
 55     });
 56   }
 57 );
 58 /**
 59  * This is the exported PetStore proxy that can be used by any server-side JavaScript code.
 60  * @exports PetStore
 61  */
 62 PetStore = getSwaggerProxy(swaggerSpecURL);
 63 
 64 /**
 65  * The PetStore.sync name space.
 66  * Contains synchronous representations of needed
 67  * asynchronous methods of the PetStore.
 68  * @namespace PetStore.sync
 69  * @memberof PetStore
 70  */
 71 PetStore.sync = {};
 72   /**
 73    * Get a pet by its nrecord id.
 74    * @name getPetById
 75    * @memberof PetStore.sync
 76    * @function
 77    * @param  {integer} petNum
 78    * @returns {json}
 79    */
 80 PetStore.sync.getPetById = Meteor.wrapAsync(
 81   function wrpr(args, headers, callback) {
 82     PetStore.pet.getPetById( args, headers
 83       , function suxs( theResult ) {
 84         Logger.info(  'Got pet #' + args.petId  );
 85         callback(null, theResult);
 86       }
 87       , function errs( theError ) {
 88         Logger.error('For Id #' + args.petId + ' : ' + theError.statusText);
 89         callback(null, theError);
 90       }
 91     );
 92   }
 93 );
 94 
 95 /**
 96  * Exposure to the Meteor name space.
 97  * This member of the Meteor namespace groups together the methods that this package will expose to client-side JavaScript
 98  * @namespace Meteor.package-ci
 99  */
100 Meteor.methods({
101   /**
102    * Get a pet by its nrecord id.
103    * @name getPetById
104    * @memberof Meteor.package-ci
105    * @method
106    * @param  {integer} petNum The PetStore demo API seems to permit full CRUD authority to anonymous users, so we use a couple of very large numbers to minimize the risk of spurious test failure.
107    * @returns {json}
108    */
109   package-ci_getPetById: function getPetById(petNum) {
110     var aPetVO = PetStore.sync.getPetById(
111       { petId: petNum},
112       {responseContentType: 'application/json'}
113     );
114     return aPetVO.status === 404
115     ? {name: JSON.parse(aPetVO.data).message}
116     : aPetVO.obj;
117   },
118 });
