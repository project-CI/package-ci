 41 /**
 42  * Can we retrieve a pet by its ID number?
 43  * @name obtainPetById
 44  * @memberof Tinytest
 45  * @function
 46  * @param  test {Tinytest} Pet #petNum is : expected
 47  * @return {None}
 48  */
 49 const petNum = 6133627028;
 50 const expected = 'Your fluffy little wolverine.';
 51 Tinytest.add('Pet #' + petNum + ' is : ' + expected, function obtainPetById(test) {
 52   var aPet = PetStore.sync.getPetById(
 53     { petId: petNum}, {responseContentType: 'application/json'}
 54   );
 55   test.equal(aPet.obj.name, expected);
 56 });
